    unfold Int64.and.
    change (Int64.unsigned (Int64.repr 255)) with 255%Z.
    assert (Hc_le: (0 <= Z.land (Int64.unsigned c) 255 <= 255)%Z). {

      lia.

      rewrite <- Z.land_assoc.
      rewrite Z.land_diag.

      unfold step2; forward_star.

          rewrite Z.land_diag.

        * split; [auto|].
          apply unmodifies_effect_refl.
