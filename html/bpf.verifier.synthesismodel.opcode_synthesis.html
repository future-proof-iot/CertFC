Module opcode_synthesis


Inductive opcode :=
  | ALU64 (* 0xX7 / 0xXf *)
  | ALU32 (* 0xX4 / 0xXc *)
  | Branch (* 0xX5 / 0xXd *)
  | LD_IMM (* 0xX8 *)
  | LD_REG (* 0xX1/0xX9 *)
  | ST_IMM (* 0xX2/0xXa *)
  | ST_REG (* 0xX3/0xXb *)
  | ILLEGAL.

Definition opcode_eqb (x y: opcode) :=
  match x, y with
  | ALU64, ALU64
  | ALU32, ALU32
  | Branch, Branch
  | LD_IMM, LD_IMM
  | LD_REG, LD_REG
  | ST_IMM, ST_IMM
  | ST_REG, ST_REG
  | ILLEGAL,ILLEGAL => true
  | _, _ => false
  end.

Definition nat_to_opcode (op: nat): opcode :=
  let opc := Nat.land op 0x07 in (* masking operation *)
    match opc with
    | 0x07 => ALU64
    | 0x04 => ALU32
    | 0x05 => Branch
    | 0x00 => LD_IMM
    | 0x01 => LD_REG
    | 0x02 => ST_IMM
    | 0x03 => ST_REG
    | _ => ILLEGAL
    end.
